                        const button = document.createElement('button');

                        const tenseName = tenseData[tenseKey].name.replace(/(Past|Present|Future)\s/, '');

                        button.dataset.tense = tenseKey;

                        button.className = 'tense-btn w-full text-left p-4 bg-white dark:bg-gray-800 rounded-xl shadow-md hover:bg-blue-50 dark:hover:bg-gray-700 transition-all-ease';

                        

                        const span = document.createElement('span');

                        span.className = 'font-semibold text-gray-700 dark:text-gray-200';

                        span.textContent = tenseName;

                        button.appendChild(span);

                        colDiv.appendChild(button);

                    });

                    tenseGrid.appendChild(colDiv);

                }

                tenseButtons = document.querySelectorAll('.tense-btn');

            }


            function updateDisplay(tenseKey, form) {

                currentTenseKey = tenseKey;

                currentForm = form;

                const data = tenseData[tenseKey];

                if (!data) return;


                tenseNameEl.textContent = data.name;

                tenseExampleEl.innerHTML = `"${data[form].example}"`;

                tenseExplanationEl.textContent = data.explanation;

                tenseStructureEl.textContent = data[form].structure;

                

                geminiResultsArea.style.display = 'none';

                geminiContent.innerHTML = '';

            }


            function updateTimeline(tenseKey) {

                const data = tenseData[tenseKey]?.timeline;

                if (!data) return;

                

                timelineMarkersContainer.innerHTML = '';

                timelineIndicator.innerHTML = ''; // Clear previous labels

                

                timelineIndicator.className = 'timeline-indicator absolute';

                timelineIndicator.classList.add(`timeline-indicator-${data.shape}`);


                if(data.type === 'point') {

                    timelineIndicator.style.left = `${data.position}%`;

                    timelineIndicator.style.width = '12px';

                } else if (data.type === 'duration') {

                    const width = data.end - data.start;

                    timelineIndicator.style.left = `${data.start + width / 2}%`;

                    timelineIndicator.style.width = `${width}%`;

                }

                

                const mainClauseText = tenseData[tenseKey]?.[currentForm]?.mainClause;

                if (mainClauseText && data.type === 'duration') {

                    const clauseLabel = document.createElement('span');

                    clauseLabel.className = 'timeline-clause-label';

                    clauseLabel.textContent = mainClauseText;

                    timelineIndicator.appendChild(clauseLabel);

                }


                if (data.marker) {

                    const markerWrapper = document.createElement('div');

                    markerWrapper.className = 'absolute top-0 h-full';

                    markerWrapper.style.left = `${data.marker.position}%`;


                    const markerLine = document.createElement('div');

                    markerLine.className = 'absolute top-1/2 -translate-y-1/2 w-px h-4 bg-gray-400 dark:bg-gray-500 -translate-x-1/2';


                    const markerLabel = document.createElement('span');

                    markerLabel.className = 'absolute top-full text-xs text-gray-500 dark:text-gray-400 whitespace-nowrap mt-2 bg-white/50 dark:bg-gray-800/50 px-1 rounded -translate-x-1/2';

                    markerLabel.textContent = data.marker.label;

                    

                    markerWrapper.appendChild(markerLine);

                    markerWrapper.appendChild(markerLabel);

                    timelineMarkersContainer.appendChild(markerWrapper);

                }


                // Highlight the correct time marker (Past, Present, Future)

                timelinePastEl.classList.remove('timeline-marker-active');

                timelinePresentEl.classList.remove('timeline-marker-active');

                timelineFutureEl.classList.remove('timeline-marker-active');


                if (tenseKey.startsWith('past')) {

                    timelinePastEl.classList.add('timeline-marker-active');

                } else if (tenseKey.startsWith('present')) {

                    timelinePresentEl.classList.add('timeline-marker-active');

                } else if (tenseKey.startsWith('future')) {

                    timelineFutureEl.classList.add('timeline-marker-active');

                }

            }


            function updateActiveTenseButton(activeButton) {

                tenseButtons.forEach(button => {

                    button.classList.remove('active', 'bg-blue-500', 'dark:bg-blue-600', 'text-white');

                    button.classList.add('bg-white', 'dark:bg-gray-800');

                    button.querySelector('span').classList.add('text-gray-700', 'dark:text-gray-200');

                });

                

                activeButton.classList.add('active', 'bg-blue-500', 'dark:bg-blue-600', 'text-white');

                activeButton.classList.remove('bg-white', 'dark:bg-gray-800');

                activeButton.querySelector('span').classList.remove('text-gray-700', 'dark:text-gray-200');

            }


            function updateActiveFormButton(activeButton) {

                formButtons.forEach(btn => {

                    btn.classList.remove('active-positive', 'active-negative', 'active-question');

                    btn.classList.add('bg-white', 'dark:bg-gray-700', 'hover:bg-gray-200', 'dark:hover:bg-gray-600');

                });

                activeButton.classList.remove('bg-white', 'dark:bg-gray-700', 'hover:bg-gray-200', 'dark:hover:bg-gray-600');

                activeButton.classList.add(`active-${currentForm}`);

            }


            function updatePageTheme(form) {

                 const body = document.body;

                 const subtitle = document.getElementById('header-subtitle');


                 body.classList.remove('bg-gray-100', 'dark:bg-gray-900', 'bg-red-50', 'dark:bg-red-900/50', 'bg-yellow-50', 'dark:bg-yellow-900/50');

                 subtitle.classList.remove('text-gray-600', 'dark:text-gray-400', 'text-gray-900', 'dark:text-gray-100');

                 

                 switch(form) {

                    case 'negative':

                        body.classList.add('bg-red-50', 'dark:bg-red-900/50');

                        subtitle.classList.add('text-gray-900', 'dark:text-gray-100');

                        break;

                    case 'question':

                        body.classList.add('bg-yellow-50', 'dark:bg-yellow-900/50');

                        subtitle.classList.add('text-gray-900', 'dark:text-gray-100');

                        break;

                    case 'positive':

                    default:

                         body.classList.add('bg-gray-100', 'dark:bg-gray-900');

                         subtitle.classList.add('text-gray-600', 'dark:text-gray-400');

                         break;

                 }

            }

            

            async function generateExamples() {

                generateBtn.disabled = true;

                geminiResultsArea.style.display = 'block';

                geminiContent.innerHTML = '<div class="w-6 h-6 rounded-full loader mx-auto"></div>';


                const tenseName = tenseData[currentTenseKey].name;

                const systemPrompt = "You are an expert English grammar teacher. Your goal is to provide clear and diverse examples.";

                const userQuery = `Generate 3 distinct and clear example sentences for the "${tenseName}" tense in the ${currentForm} form. The sentences should cover different subjects and contexts. Format the response as a numbered list.`;


                const apiKey = ""; 

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;


                const payload = {

                    contents: [{ parts: [{ text: userQuery }] }],

                    systemInstruction: { parts: [{ text: systemPrompt }] },

                };


                try {

                    let response;

                    for (let i = 0; i < 3; i++) {

                        response = await fetch(apiUrl, {

                            method: 'POST',

                            headers: { 'Content-Type': 'application/json' },

                            body: JSON.stringify(payload)

                        });

                        if (response.ok) break;

                        await new Promise(resolve => setTimeout(resolve, 1000 * Math.pow(2, i)));

                    }

                    

                    if (!response.ok) throw new Error(`API request failed with status ${response.status}`);


                    const result = await response.json();

                    const text = result.candidates?.[0]?.content?.parts?.[0]?.text;


                    if (text) {

                        geminiContent.innerHTML = `<ul class="list-disc list-inside space-y-2 text-gray-600 dark:text-gray-300">${text.split(/\d+\.\s/).filter(s => s.trim()).map(s => `<li>${s.trim()}</li>`).join('')}</ul>`;

                    } else {

                        throw new Error('No content received from API.');

                    }

                } catch (error) {

                    console.error("Gemini API call failed:", error);

                    geminiContent.innerHTML = `<p class="text-red-500 text-center">Sorry, something went wrong while generating examples. Please try again later.</p>`;

                } finally {

                    generateBtn.disabled = false;

                }

            }

            

            function initialize() {

                buildTenseGrid();

                

                const initialActiveButton = document.querySelector(`[data-tense="${currentTenseKey}"]`);

                updateActiveTenseButton(initialActiveButton);

                updateDisplay(currentTenseKey, currentForm);

                updateTimeline(currentTenseKey);


                tenseButtons.forEach(button => {

                    button.addEventListener('click', () => {

                        const tenseKey = button.dataset.tense;

                        updateDisplay(tenseKey, currentForm);

                        updateActiveTenseButton(button);

                        updateTimeline(tenseKey);

                    });

                });


                formButtons.forEach(button => {

                    button.addEventListener('click', () => {

                        const form = button.dataset.form;

                        currentForm = form;

                        updateDisplay(currentTenseKey, currentForm);

                        updateActiveFormButton(button);

                        updatePageTheme(form);

                        updateTimeline(currentTenseKey); // Re-render timeline to update clause label

                    });

                });

                

                generateBtn.addEventListener('click', generateExamples);

            }


            initialize();

        });
